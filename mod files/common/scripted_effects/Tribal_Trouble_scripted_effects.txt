break_cemented_non_aggression_pact = {
	$FIRST$ = {
		every_relation = {
			type = non_aggression_pactee
			limit = {
				OR = {
					THIS = $SECOND$
					is_close_family_of = $SECOND$
				}
				NOT = { is_in_tribal_non_agression_pact_with = { TARGET = PREV }}
			}
			remove_relation_non_aggression_pactee = PREV
		}
		every_close_family_member = {
			every_relation = {
				type = non_aggression_pactee
				limit = {
					OR = {
						THIS = $SECOND$
						is_close_family_of = $SECOND$
					}
					NOT = { is_in_tribal_non_agression_pact_with = { TARGET = PREV }}
				}
				remove_relation_non_aggression_pactee = PREV
			}
		}
	}
}
